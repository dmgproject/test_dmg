drop table NOTICE;
CREATE SEQUENCE SEQ_NNO
START WITH 1
INCREMENT BY 1
NOCYCLE
NOCACHE;
CREATE TABLE NOTICE (
      NNO NUMBER PRIMARY KEY,
      NTITLE VARCHAR2(500) NOT NULL,
      NCONTENT VARCHAR2(3000) NOT NULL,
      NWRITER VARCHAR2(15) NOT NULL,
      NCOUNT NUMBER DEFAULT 0,
      NBOARDFILE VARCHAR2(600),
      NDATE DATE DEFAULT SYSDATE,
      CONSTRAINT FK_NWRITER FOREIGN KEY(NWRITER) REFERENCES MEMBER(USERID) ON DELETE SET NULL
);
CREATE OR REPLACE PROCEDURE TEST_NOTICE_DATA AS
BEGIN 
 FOR CNT IN 1..250 LOOP
    INSERT INTO NOTICE VALUES (SEQ_NNO.NEXTVAL,'TEST SAMPLE ' || SEQ_NNO.CURRVAL, 'TEST SAMPLE 데이터입니다. ' || SEQ_NNO.CURRVAL, 'admin', DEFAULT, null, TO_DATE('19/01/01')+SEQ_NNO.CURRVAL);
 END LOOP;
 COMMIT;
END;
/

EXEC TEST_NOTICE_DATA;

COMMIT;
