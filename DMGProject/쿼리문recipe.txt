CREATE TABLE RECIPE(
  RENUM  NUMBER PRIMARY KEY,
  REMAIN VARCHAR2(30) NOT NULL,
  RESUB VARCHAR2(2000) NOT NULL,
  RE_NAME VARCHAR2(50) NOT NULL,
  RECIPE VARCHAR2(3000) NOT NULL,
  RECIPEFILE VARCHAR2(600),
  UPLOADDATE DATE DEFAULT SYSDATE,
  COUNT NUMBER
);
);

CREATE TABLE ATTACHMENT(
    FNO NUMBER PRIMARY KEY,
    RENUM NUMBER NOT NULL,
    ORIGINNAME VARCHAR2(2000) NOT NULL,
    CHANGENAME VARCHAR2(2000) NOT NULL,
    FILEPATH VARCHAR2(2000),
    UPLOADDATE DATE DEFAULT SYSDATE,
    CONSTRAINT FK_ATT_RENUM FOREIGN KEY(RENUM) REFERENCES RECIPE(RENUM)
);

CREATE TABLE MATERIAL 
(
  RENO NUMBER NOT NULL, 
  RENUM NUMBER, 
  RESUB VARCHAR2(50 BYTE) NOT NULL, 
  RECOUNT VARCHAR2(50 BYTE) NOT NULL, 
  CONSTRAINT FK_MTR_RENUM FOREIGN KEY(RENUM) REFERENCES RECIPE(RENUM)
)

CREATE SEQUENCE SEQ_RENUM START WITH 1 INCREMENT BY 1 NOCYCLE NOCACHE;
CREATE SEQUENCE SEQ_FNO START WITH 1 INCREMENT BY 1 NOCYCLE NOCACHE;
CREATE SEQUENCE SEQ_RENO START WITH 1 INCREMENT BY 1 NOCYCLE NOCACHE;
