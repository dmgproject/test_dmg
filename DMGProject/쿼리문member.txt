drop table notice;
drop table member;
create table MEMBER(
    USERID VARCHAR2(30) PRIMARY KEY,
    PASSWORD VARCHAR2(100) NOT NULL,
    USERNAME VARCHAR2(30) NOT NULL,
    EMAIL VARCHAR2(30),
    PHONE VARCHAR2(14),
    ADDRESS VARCHAR2(300),
    ENROLLDATE DATE DEFAULT SYSDATE,
    STATUS CHAR(1) DEFAULT 'Y' CHECK(STATUS IN ('Y', 'N', 'A'))
);
