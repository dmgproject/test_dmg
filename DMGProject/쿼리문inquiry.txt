CREATE TABLE INQUIRY (
      INO NUMBER PRIMARY KEY,
      ITYPE NUMBER NOT NULL,
      ITITLE VARCHAR2(500) NOT NULL,
      ICONTENT VARCHAR2(3000) NOT NULL,
      IWRITER VARCHAR2(15) NOT NULL,
      ICOUNT NUMBER DEFAULT 0,
      INQUIRYFILE VARCHAR2(600),
      IDATE DATE DEFAULT SYSDATE,
      STATUS VARCHAR2(1) DEFAULT 'Y' CHECK (STATUS IN ('Y', 'N')),
      CONSTRAINT FK_ITYPE FOREIGN KEY(ITYPE) REFERENCES CATEGORY(INQUIRYTYPE),
      CONSTRAINT FK_IWRITER FOREIGN KEY(IWRITER) REFERENCES MEMBER(USERID) ON DELETE SET NULL
);
